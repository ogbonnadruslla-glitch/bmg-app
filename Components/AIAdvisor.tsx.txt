import React, { useState, useCallback, useRef, useEffect } from 'react';
import { SalesTarget, SalesLog, Message } from '../types';
import { getBusinessAdvice, startReviewCheckin } from '../services/geminiService';

const Spinner: React.FC<{ color?: 'white' | 'gray' }> = ({ color = 'white' }) => (
    <div className={`animate-spin rounded-full h-5 w-5 border-b-2 ${color === 'white' ? 'border-white' : 'border-gray-600'}`}></div>
);

type AdvisorMode = 'IDEAS' | 'ADVICE' | 'HUB' | 'REVIEW';

const infoHubTopics = [
    "Effective Marketing Strategies for Small Businesses",
    "Improving Customer Retention and Loyalty",
    "Financial Management and Budgeting Tips",
    "Scaling Operations for Growth",
    "Building a Strong Brand Identity",
    "Leveraging Social Media for Sales",
];

const AIAdvisor: React.FC<{ targets: SalesTarget; logs: SalesLog[] }> = ({ targets, logs }) => {
    const [mode, setMode] = useState<AdvisorMode>('IDEAS');
    const [messages, setMessages] = useState<Message[]>([]);
    const [userInput, setUserInput] = useState('');
    const [isLoading, setIsLoading] = useState(false);
    const chatContainerRef = useRef<HTMLDivElement>(null);
    
    useEffect(() => {
        chatContainerRef.current?.scrollTo(0, chatContainerRef.current.scrollHeight);
    }, [messages]);

    const handleModeChange = useCallback((newMode: AdvisorMode) => {
        setMode(newMode);
        setMessages([]);
        setUserInput('');
    }, []);
    
    const handleStartReview = useCallback(async () => {
        setIsLoading(true);
        setMessages([]);
        const openingMessage = await startReviewCheckin(targets, logs);
        setMessages([{ sender: 'ai', text: openingMessage }]);
        setIsLoading(false);
    }, [targets, logs]);

    const handleFetchTopic = useCallback(async (topic: string) => {
        setIsLoading(true);
        setMessages([]);
        const advice = await getBusinessAdvice(topic);
        setMessages([{ sender: 'ai', text: advice }]);
        setIsLoading(false);
    }, []);

    const handleSubmit = async (e: React.FormEvent) => {
        e.preventDefault();
        if (!userInput.trim() || isLoading) return;

        const newMessages: Message[] = [...messages, { sender: 'user', text: userInput }];
        setMessages(newMessages);
        setUserInput('');
        setIsLoading(true);

        const aiResponse = await getBusinessAdvice(userInput);

        setMessages([...newMessages, { sender: 'ai', text: aiResponse }]);
        setIsLoading(false);
    };
    
    const renderModeContent = () => {
        switch (mode) {
            case 'HUB':
                return (
                    <div className="p-4 md:p-6">
                        <h2 className="text-xl font-semibold mb-4 text-gray-900">Information Hub</h2>
                        <p className="text-gray-500 mb-6">Select a category to receive information and advice.</p>
                        <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                            {infoHubTopics.map(topic => (
                                <button key={topic} onClick={() => handleFetchTopic(topic)} className="p-4 bg-gray-50 hover:bg-gray-100 border border-gray-200 rounded-lg text-left transition-colors text-gray-700">
                                    {topic}
                                </button>
                            ))}
                        </div>
                        {isLoading && <div className="mt-6 flex justify-center"><Spinner color="gray" /></div>}
                        {messages.length > 0 && (
                             <div className="mt-6 prose max-w-none bg-gray-50 border border-gray-200 rounded-lg p-6">
                                <pre className="whitespace-pre-wrap font-sans text-sm text-gray-700">{messages[0].text}</pre>
                            </div>
                        )}
                    </div>
                );
            case 'REVIEW':
                 return (
                    <div className="flex flex-col h-full">
                        <div className="p-4 text-center">
                            <h2 className="text-xl font-semibold mb-2 text-gray-900">Weekly Review Check-in</h2>
                            <p className="text-gray-500 mb-4">Let's review your progress with your AI business coach.</p>
                            <button onClick={handleStartReview} disabled={isLoading} className="px-6 py-2 bg-green-600 hover:bg-green-700 text-white font-semibold rounded-lg shadow-md transition-colors flex items-center justify-center space-x-2 mx-auto disabled:bg-green-300">
                                {isLoading ? <Spinner /> : null}
                                <span>Start Check-in</span>
                            </button>
                        </div>
                        {messages.length > 0 && renderChatInterface()}
                    </div>
                );
            default:
                return renderChatInterface();
        }
    };
    
    const renderChatInterface = () => (
        <div className="flex flex-col h-full">
            <div ref={chatContainerRef} className="flex-1 overflow-y-auto p-4 space-y-4">
                {messages.map((msg, index) => (
                    <div key={index} className={`flex ${msg.sender === 'user' ? 'justify-end' : 'justify-start'}`}>
                        <div className={`max-w-lg p-3 rounded-xl shadow-sm ${msg.sender === 'user' ? 'bg-green-600 text-white' : 'bg-gray-100 text-gray-800'}`}>
                            <pre className="whitespace-pre-wrap font-sans">{msg.text}</pre>
                        </div>
                    </div>
                ))}
                {isLoading && (
                     <div className="flex justify-start">
                        <div className="max-w-lg p-3 rounded-xl bg-gray-100 text-gray-600 flex items-center">
                            <Spinner color="gray" /> <span className="ml-2">Thinking...</span>
                        </div>
                    </div>
                )}
            </div>
            <div className="p-4 border-t border-gray-200 bg-white">
                <form onSubmit={handleSubmit} className="flex items-center space-x-2 sm:space-x-4">
                    <input
                        type="text"
                        value={userInput}
                        onChange={(e) => setUserInput(e.target.value)}
                        placeholder={
                            mode === 'IDEAS' ? "Ask for growth ideas..." :
                            mode === 'ADVICE' ? "Ask for specific business advice..." :
                            "Type your response..."
                        }
                        className="flex-1 bg-gray-100 border-gray-200 text-gray-900 rounded-lg px-4 py-2 focus:ring-green-500 focus:border-green-500"
                    />
                    <button type="submit" disabled={isLoading} className="p-2 bg-green-600 hover:bg-green-700 rounded-full text-white disabled:bg-green-300">
                        <svg xmlns="http://www.w3.org/2000/svg" className="h-6 w-6" fill="none" viewBox="0 0 24 24" stroke="currentColor"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M5 13l4 4L19 7" /></svg>
                    </button>
                </form>
            </div>
        </div>
    );
    
    return (
        <div className="space-y-6 md:space-y-8 h-full flex flex-col">
            <h1 className="text-3xl font-bold text-gray-900 px-2">BMG AI Advisor</h1>
            <div className="border-b border-gray-200">
                <nav className="-mb-px flex space-x-8" aria-label="Tabs">
                    <button onClick={() => handleModeChange('IDEAS')} className={`${mode === 'IDEAS' ? 'border-yellow-500 text-yellow-600' : 'border-transparent text-gray-500 hover:text-gray-700 hover:border-gray-300'} whitespace-nowrap py-4 px-1 border-b-2 font-medium text-sm`}>Growth Ideas</button>
                    <button onClick={() => handleModeChange('ADVICE')} className={`${mode === 'ADVICE' ? 'border-yellow-500 text-yellow-600' : 'border-transparent text-gray-500 hover:text-gray-700 hover:border-gray-300'} whitespace-nowrap py-4 px-1 border-b-2 font-medium text-sm`}>Business Advice</button>
                    <button onClick={() => handleModeChange('HUB')} className={`${mode === 'HUB' ? 'border-yellow-500 text-yellow-600' : 'border-transparent text-gray-500 hover:text-gray-700 hover:border-gray-300'} whitespace-nowrap py-4 px-1 border-b-2 font-medium text-sm`}>Info Hub</button>
                    <button onClick={() => handleModeChange('REVIEW')} className={`${mode === 'REVIEW' ? 'border-yellow-500 text-yellow-600' : 'border-transparent text-gray-500 hover:text-gray-700 hover:border-gray-300'} whitespace-nowrap py-4 px-1 border-b-2 font-medium text-sm`}>Review Check-in</button>
                </nav>
            </div>
            <div className="bg-white rounded-xl border border-gray-200 shadow-sm flex-1 min-h-0">
                {renderModeContent()}
            </div>
        </div>
    );
};

export default AIAdvisor;
