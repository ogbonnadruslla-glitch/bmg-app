export type Currency = 'NGN' | 'USD' | 'EUR' | 'GBP';

export interface SalesTarget {
  daily: number;
  weekly: number;
  monthly: number;
  quarterly: number;
  annually: number;
}

export interface SalesLog {
  id: string;
  date: string; // YYYY-MM-DD
  amount: number;
  notes: string;
  category?: string;
  product?: string;
}

export interface Message {
    sender: 'user' | 'ai' | 'support';
    text: string;
    timestamp?: string;
}

export interface UserProfile {
    name: string;
    businessName: string;
    businessCategory: string;
    email: string;
    phone: string;
    address: string;
    state: string;
    lga: string;
    about: string;
    // Fix: Use the exported Currency type.
    currency: Currency;
}

export interface Product {
    id: string;
    name: string;
    category: string;
    price?: number;
    margin?: number;
}

export type UserPlan = 'basic' | 'pro';

export interface ProjectionDetails {
    monthlyCosts: number[];
    monthlyIncome: number[];
    monthlyDistribution: number[]; // Users
    monthlyRetailPoints: number[];
    monthlyLocations: number[];
    monthlyChannels: number[];
}

export interface DistributionTargets {
    users: number;
    retailPoints: number;
    locations: number;
    channels: number;
}

  BANNER = 'BANNER',
  POPUP = 'POPUP',
  SPONSORED_LISTING = 'SPONSORED_LISTING',
}

export interface AdContent {
  title: string;
  description: string;
  imageUrl?: string;
  videoUrl?: string;
  ctaText: string;

}

  id: string;

  placement: string; // e.g., 'dashboard-banner', 'global-popup', 'dashboard-feed'
